/* ================= HELPERS ================= */
float calculateDrift(double lat1, double lon1, double lat2, double lon2) {
  if (lat1 == 0) return 0;
  double dLat = radians(lat2 - lat1), dLon = radians(lon2 - lon1);
  double a = sin(dLat/2)*sin(dLat/2) + cos(radians(lat1))*cos(radians(lat2))*sin(dLon/2)*sin(dLon/2);
  return 6371000 * 2 * atan2(sqrt(a), sqrt(1-a));
}

void updateEWMA(float value, float alpha, float &mean, float &variance) {
  float delta = value - mean;
  mean += alpha * delta;
  variance = (1.0 - alpha) * (variance + alpha * delta * delta);
  if (variance < 0.0001) variance = 0.0001;
}

void kalmanUpdate(double mLat, double mLon, float dt) {
  if(!kalmanInitialized) { kfLat=mLat; kfLon=mLon; kalmanInitialized=true; return; }
  kfLat += kfVLat * dt; kfLon += kfVLon * dt;
  double eLat = mLat - kfLat, eLon = mLon - kfLon;
  kfLat += 0.22 * eLat; kfLon += 0.22 * eLon;
  if(dt > 0.01) { kfVLat += 0.08 * (eLat/dt); kfVLon += 0.08 * (eLon/dt); }
}

void appendAttackLog(String type) {
  unsigned long now = millis();
  if (type == lastLoggedAttack && (now - lastAttackLogMs) < 5000) return;
  int idx = (attackLogStart + attackLogCount) % MAX_ATTACK_LOGS;
  attackLogs[idx] = {now/1000, displayLat, displayLon, antiSpoofRisk, type};
  if (attackLogCount < MAX_ATTACK_LOGS) attackLogCount++; else attackLogStart = (attackLogStart + 1) % MAX_ATTACK_LOGS;
  lastLoggedAttack = type; lastAttackLogMs = now;
}

/* ================= SECURITY LOGIC ================= */
void processSecurityLogic() {
  unsigned long nowMs = millis();
  float dt = (lastFixMs == 0) ? 1.0 : constrain((nowMs - lastFixMs) / 1000.0, 0.2, 5.0);
  lastFixMs = nowMs;

  if (!gps.location.isValid() || gps.satellites.value() < 4) {
    aiStatus = "NO GPS LOCK"; antiSpoofRisk = 0.2; return;
  }

  currentLat = gps.location.lat(); currentLon = gps.location.lng();
  speedKph = gps.speed.kmph(); satellites = gps.satellites.value(); hdopValue = gps.hdop.hdop();
  
  driftMeters = calculateDrift(lastLat, lastLon, currentLat, currentLon);
  kalmanUpdate(currentLat, currentLon, dt);

  // Feature Extraction for AI
  float features[5] = {(float)currentLat, (float)currentLon, speedKph, satellites, hdopValue};
  int pred = model.predict(features);

  // EWMA Statistics
  updateEWMA(driftMeters, 0.07, driftMean, driftVar);
  float driftZ = fabs(driftMeters - driftMean) / sqrt(driftVar);

  // CRITICAL FIX: Hardware Veto Logic
  // AI Prediction (pred == 1) is only valid if Hardware quality is also low
  bool hardwareDistress = (satellites < 6 || hdopValue > 2.5);
  bool aiFlagsJamming = (pred == 1 && hardwareDistress);

  // Stability Gate Logic
  float gateLimit = max(35.0f, (speedKph * dt) + 20.0f);
  stabilityGateBlocked = (driftMeters > gateLimit) || (driftMeters > 400.0);

  if (stabilityGateBlocked) {
    displayLat = lastLat; displayLon = lastLon;
    spoofReason = "Stability Gate: Impossible jump blocked.";
    spoofCounter += 2;
  } else {
    displayLat = currentLat; displayLon = currentLon;
    lastLat = currentLat; lastLon = currentLon;
    spoofCounter = max(0, spoofCounter - 1);
  }

  antiSpoofRisk = constrain((driftZ * 0.15) + (stabilityGateBlocked ? 0.5 : 0.0), 0, 1);

  if (jamSim || aiFlagsJamming) {
    aiStatus = "JAMMING DETECTED";
    appendAttackLog("Jamming");
  } else if (spoofSim || spoofCounter > 4) {
    aiStatus = "SPOOF DETECTED";
    appendAttackLog("Spoofing");
  } else {
    aiStatus = "NORMAL";
    spoofReason = "Signal normal.";
  }

  trustScore = constrain((satellites / 12.0) * (1.0 - antiSpoofRisk), 0, 1);
}

  doc["lat"] = displayLat; doc["lon"] = displayLon; doc["status"] = aiStatus;
  doc["sat"] = satellites; doc["hdop"] = hdopValue; doc["drift"] = driftMeters;
  doc["trust"] = trustScore; doc["speed"] = speedKph; doc["risk"] = antiSpoofRisk;
  doc["gate"] = stabilityGateBlocked; doc["reason"] = spoofReason;

}

void handleLogs() {
  StaticJsonDocument<1500> doc; JsonArray l = doc.createNestedArray("logs");
  for (int i=0; i<attackLogCount; i++) {
    int idx = (attackLogStart + i) % MAX_ATTACK_LOGS;
    JsonObject r = l.createNestedObject();
    r["time"] = attackLogs[idx].sec; r["lat"] = attackLogs[idx].lat;
    r["attack"] = attackLogs[idx].attack; r["risk"] = attackLogs[idx].risk;
  }
  String j; serializeJson(doc, j); server.send(200, "application/json", j);
}

void setup() {
  Serial.begin(115200);
  gpsSerial.begin(9600, SERIAL_8N1, 16, 17);
  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) delay(500);
  server.on("/", [](){ server.send(200, "text/html", index_html); });
  server.on("/status", handleStatus);
  server.on("/logs", handleLogs);
  server.on("/spoof", [](){ spoofSim=true; server.send(200); });
  server.on("/jam", [](){ jamSim=true; server.send(200); });
  server.on("/reset", [](){ deepPurgeResync(); server.send(200); });
  server.begin();
}

void loop() {
  server.handleClient();
  while (gpsSerial.available()) gps.encode(gpsSerial.read());
  if (gps.location.isUpdated()) processSecurityLogic();
}
